  <script>
    // collapsible helpers
    document.querySelectorAll('.collapsible .content').forEach((el)=>{
      el.style.display = 'block'; // default expanded
    });

    function toggle(header){
      const article = header.closest('article');
      const content = article.querySelector('.content');
      const chev = header.querySelector('.chev');
      if(!article.classList.contains('collapsed')){
        // collapse
        content.style.display = 'none';
        article.classList.add('collapsed');
        chev.textContent = '‚ñ∂';
      } else {
        // expand
        content.style.display = 'block';
        article.classList.remove('collapsed');
        chev.textContent = '‚ñ∂';
      }
    }

    // toggle all (open/close)
    let allCollapsed = false;
    document.getElementById('toggleAll').addEventListener('click',()=>{
      document.querySelectorAll('.collapsible').forEach((a)=>{
        const content = a.querySelector('.content');
        const chev = a.querySelector('.chev');
        if(!allCollapsed){
          content.style.display = 'none';
          a.classList.add('collapsed');
          chev.textContent = '‚ñ∂';
        } else {
          content.style.display = 'block';
          a.classList.remove('collapsed');
          chev.textContent = '‚ñ∂';
        }
      });
      allCollapsed = !allCollapsed;
      document.getElementById('toggleAll').textContent = allCollapsed ? 'Expand All Sections' : 'Collapse All Sections';
    });

    // simple client-side filter
    document.getElementById('filter').addEventListener('input', function(){
      const q = this.value.trim().toLowerCase();
      document.querySelectorAll('article.card').forEach((article)=>{
        const text = article.innerText.toLowerCase();
        if(q === '' || text.includes(q)){
          article.style.display = 'block';
        } else {
          article.style.display = 'none';
        }
      });
    });
  </script>
